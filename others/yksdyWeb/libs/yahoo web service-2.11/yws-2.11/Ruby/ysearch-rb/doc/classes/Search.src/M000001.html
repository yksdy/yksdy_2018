    def initialize(app_id)
        @NAME = "Search"
        @SERVICE = "Search"
        @PROTOCOL = "http"
        @SERVER = "api.search.yahoo.com"
        @VERSION = "V1"
        @NEXT_QID = 1

        @service = { "name" => @NAME,
                      "protocol" => @PROTOCOL,
                      "server" => @SERVER,
                      "version" => @VERSION,
                      "service" => @SERVICE }

        @app_id = app_id
        @valid_params = {}

        @qid = @NEXT_QID
        @NEXT_QID = @NEXT_QID + 1

        init_valid_params()
        reset()
    end
