    def initialize(app_id, tag, yahooid='', sort='date', reverse_sort=0, results=10, start=1)
        super Search
        set_app_id(app_id)
        set_svc_service($SERVICES['myweb2_url'][0])
        set_svc_name($SERVICES['myweb2_url'][1])
        set_tag(tag)
        set_yid(yahooid)
        set_sort(sort)
        set_reverse_sort(reverse_sort)
        set_results(results)
        set_start(start)
    end
